  321.     /**

  326.      */
  327.     public function route($route = NULL)
  328.     {
  329.         //Changes to true when a valid page can be found.
  330.         $dispatch = false; 
  331.         
  332.         //First determine which routing information to use
  333.         if(!is_null($route))                                //Use the supplied Route
  334.         {
  335.             self::$route = $route;
  336.             
  337.         }
  338.         elseif(array_key_exists('PATH_INFO', $_SERVER))        //Use the URI route
  339.         {
  340.             self::$route = $_SERVER['PATH_INFO']; 
  341.             self::$route = urldecode(self::$route);            //URL decode any special characters
  342.         }
  343.         else                                                //Use the default route
  344.         {
  345.             self::$route = 'index/index';
  346.         }
  347.         
  348.         //Run some route cleaning operations.
  349.         self::$route = str_replace('\\','/',self::$route);            //Convert backslashes to forward slashes 
  350.         if(substr(self::$route, 0, 1) == '/')                        //Remove a starting forward slash
  351.         {
  352.             self::$route = substr(self::$route, 1, strlen(self::$route)-1);
  353.         }
  354.         if(($end = strpos(self::$route,'.')) !== false)                //End routing information on the first "." occurance
  355.         {
  356.             self::$route = substr(self::$route, 0, $end);
  357.         }
  358.         
  359.         //echo '<p>Current Route: '.self::$route.'</p>';
  360.         
  361.         //Check to see if a script exists with that route.
  362.         $scriptRoute = SCRIPT_ROOT.self::$route.'.php';
  363.         if(file_exists($scriptRoute))
  364.         {
  365.             //Check for valid path information
  366.             if(ctype_alnum(str_replace(array('/','_','-'),'',self::$route)))
  367.             {
  368.                 //Authentication Check
  369.                 if(Staple_Config::getValue('auth', 'enabled') != 0)
  370.                 {
  371.                     //Check for an excluded script route
  372.                     $allowedScripts = (array)Staple_Config::getValue('auth', 'allowedRoute');
  373.                     if(in_array(self::$route, $allowedScripts) === true)
  374.                     {
  375.                         //Script does not require auth, Dispatch Script
  376.                         $this->dispatchScript($scriptRoute);
  377.                     }
  378.                     else
  379.                     {
  380.                         //Check for a login
  381.                         if($this->auth->isAuthed() === true)
  382.                         {
  383.                             //Valid login found, Dispatch Script
  384.                             $this->dispatchScript($scriptRoute);
  385.                         }
  386.                         else
  387.                         {
  388.                             //No valid login, no Auth
  389.                             $this->auth->noAuth();
  390.                         }
  391.                     }
  392.                 }
  393.                 else
  394.                 {
  395.                     //Auth Disabled, Dispatch Script
  396.                     $this->dispatchScript($scriptRoute);
  397.                 }
  398.                 $dispatch = true;
  399.             }
  400.         }
  401.         else
  402.         {
  403.             //No Script found, routing to controller/action
  404.             
  405.             //Split the route into it's component elements.
  406.             $splitRoute = explode('/',self::$route);
  407.             
  408.             //If the route only contains a controller add the index action
  409.             if(count($splitRoute) == 1)

  411.                 array_push($splitRoute, 'index');
  412.             }
  413.             elseif(count($splitRoute) >= 2)
  414.             {
  415.                 //Correct for extra ending slash.
  416.                 if(strlen($splitRoute[1]) < 1)
  417.                 {
  418.                     $splitRoute[1] = 'index';
  419.                 }
  420.                 //If the action is numeric, it is not the action. Insert the index action into the route.
  421.                 if(is_numeric($splitRoute[1]))
  422.                 {
  423.                     $shift = array_shift($splitRoute);
  424.                     array_unshift($splitRoute, $shift, 'index');
  425.                 }
  426.             }
  427.             $class = Staple_Link::methodCase(array_shift($splitRoute));
  428.             $method = Staple_Link::methodCase(array_shift($splitRoute));
  429.             if(ctype_alnum($class) && ctype_alnum($method))
  430.             {
  431.                 $dispatchClass = $class.'Controller';
  432.                 $started = false;
  433.             
  434.                 //Check for the controller existence
  435.                 if(class_exists($dispatchClass))
  436.                 {
  437.                     //Check for the action existence
  438.                     if(method_exists($dispatchClass, $method))
  439.                     {
  440.                         //If the controller has not been created yet, create an instance and store it in the front controller
  441.                         if(!array_key_exists($class, self::$controllers))
  442.                         {
  443.                             self::$controllers[$class] = new $dispatchClass();
  444.                             self::$controllers[$class]->_start();
  445.                             $started = true;
  446.                         }
  447.                         
  448.                         //Verify that an instance of the controller class exists and is of the right type
  449.                         if(self::$controllers[$class] instanceof Staple_Controller)
  450.                         {
  451.                             //Check if global Auth is enabled.
  452.                             if(Staple_Config::getValue('auth', 'enabled') != 0)
  453.                             {
  454.                                 if(self::$controllers[$class]->_auth($method) && self::$controllers[$class]->_authLevel($method) == 0)
  455.                                 {
  456.                                     $this->dispatchController($class, $method, $splitRoute, $started);
  457.                                 }
  458.                                 else
  459.                                 {
  460.                                     if($this->auth->isAuthed() && $this->auth->getAuthLevel() >= self::$controllers[$class]->_authLevel($method))
  461.                                     {
  462.                                         $this->dispatchController($class, $method, $splitRoute, $started);
  463.                                     }
  464.                                     else 
  465.                                     {
  466.                                         $this->auth->noAuth();
